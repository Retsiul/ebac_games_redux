       setupTests.ts

__tests__ com o arquivo NomeDoComponente.test.tsx

import { Provider } from 'react-redux'

   render(

    )

°teste de elementos com valores interativos que armazenam dados
como é com redux,então iremos partir do slice de um elemento
       criamos uma pasta chamada
  utils/tests.tsx(componente de teste)
  criamos uma função que renderiza os componentes na memória(um render customizado)
   sintaxe

 import { RenderOptions } from '@testing-library/react'
import { PreloadedState } from '@reduxjs/toolkit'
import React from 'react'

type ExtendenRenderOptions = Omit<RenderOptions, 'queries'>{
  preloadedState?:PreloadedState
}

export function renderizaComProvider(
  elemento: React.ReactElement,
  {
    preloadedState = {},
    store,
    ...opcoesAdicionais
  }: ExtendenRenderOptions = {}
)

   elemento um objeto e é o argumento da função,
   React.ReactElement é o tipo
   preloadedState:{} é o estado inicial do elemento
   store é a store quem vem do slice
   opcoesAdicionais é um argumento adicional
   : ExtendenRenderOptions = {} é o tipo e também um objeto
   Omit é um tipo do type scrip,ele serve para omitir propriedades obregatoria de um
   objeto
   RenderOptions é um tipo e também um objeto
   estamos omitindo a propriedade 'queries'
   muito desse tipos são objetos pois tem funções imbutidas

°muito importante quando se trabalha com testes que querem acessar
dados do slice é preciso criar uma configuração especifica na store,
a configuração é bem complexa,nesse caso as configurações se encontrar na documentação
tem que ver a documentação do redux e do react
